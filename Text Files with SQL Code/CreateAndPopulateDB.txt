CREATE TABLE User(
	UserID INT NOT NULL,
	Name VARCHAR(30) NOT NULL,
	Email VARCHAR(30) NOT NULL,
	Password VARCHAR(30) NOT NULL,
	Paypal VARCHAR(30),
	Venmo VARCHAR(30),
	Karma INT,
	Cryptocurrency VARCHAR(30),
	PRIMARY KEY(UserID)
);
CREATE TABLE Buyer(
	BuyerID INT NOT NULL,
	PRIMARY KEY(BuyerID),
	FOREIGN KEY(BuyerID) REFERENCES User(UserID) ON DELETE SET NULL ON UPDATE CASCADE
);
CREATE TABLE Seller(
	SellerID INT NOT NULL,
	Bio VARCHAR(500),
	PRIMARY KEY(SellerID),
	FOREIGN KEY(SellerID) REFERENCES User(UserID) ON DELETE SET NULL ON UPDATE CASCADE
);
CREATE TABLE CreditDebit(
	UserID INT NOT NULL,
	CardNumber INT NOT NULL,
	CVV INT NOT NULL,
	CardholderName VARCHAR(30) NOT NULL,
	ExpDate DATE NULL,
	PRIMARY KEY(UserID, CardNumber),
	FOREIGN KEY(UserID) REFERENCES User(UserID) ON DELETE SET NULL ON UPDATE CASCADE
);
CREATE TABLE ShoppingCart(
	ShoppingCartID INT NOT NULL,
	PRIMARY KEY(ShoppingCartID),
	FOREIGN KEY(ShoppingCartID) REFERENCES User(UserID) ON DELETE CASCADE ON UPDATE CASCADE
);
CREATE TABLE ShoppingCartProduct(
	ShoppingCartID INT NOT NULL,
	ProductID INT NOT NULL,
	PRIMARY KEY(ShoppingCartID, ProductID),
	FOREIGN KEY(ShoppingCartID) REFERENCES ShoppingCart(ShoppingCartID) ON DELETE CASCADE  ON UPDATE CASCADE,
	FOREIGN KEY(ProductID) REFERENCES Product(ProductID) ON DELETE SET NULL ON UPDATE CASCADE
);
CREATE TABLE SellerReview(
	BuyerID INT NOT NULL,
	SellerID INT NOT NULL,
	ReviewID INT NOT NULL,
	Review VARCHAR(500),
	PRIMARY KEY(ReviewID),
	FOREIGN KEY(BuyerID) REFERENCES Buyer(BuyerID) ON DELETE SET NULL ON UPDATE CASCADE,
	FOREIGN KEY(SellerID) REFERENCES Seller(SellerID) ON DELETE SET NULL ON UPDATE CASCADE
);
CREATE TABLE SellerPhotos(
	SellerPhotoID INT NOT NULL, 
	SellerID INT, 
	Photo VARCHAR(500) NOT NULL,
	PRIMARY KEY(SellerPhotoID),
	FOREIGN KEY(SellerID) REFERENCES Seller(SellerID) ON DELETE SET NULL ON UPDATE CASCADE
);
CREATE TABLE Store(
	SellerID INT NOT NULL,
	StoreID INT NOT NULL,
	Name VARCHAR(30) NOT NULL,
	Description VARCHAR(500),
	Banner VARCHAR(30),
	PRIMARY KEY(StoreID),
	FOREIGN KEY(SellerID) REFERENCES Seller(SellerID) ON DELETE SET NULL ON UPDATE CASCADE
);
CREATE TABLE StoreAcceptedPaymentTypes(
	StoreID INT NOT NULL,
	Payment VARCHAR(15) NOT NULL,
	PRIMARY KEY(StoreID, Payment),
	FOREIGN KEY(StoreID) REFERENCES Store(StoreID) ON DELETE CASCADE ON UPDATE CASCADE
);
CREATE TABLE StoreURL(
	StoreID INT NOT NULL,
	URL VARCHAR(30) NOT NULL,
	PRIMARY KEY(StoreID, URL),
	FOREIGN KEY(StoreID) REFERENCES Store(StoreID) ON DELETE CASCADE ON UPDATE CASCADE
);
CREATE TABLE Product(  
	StoreID INT NOT NULL,
	ProductID INT NOT NULL,
	Price DECIMAL(6,2) NOT NULL,
	Title VARCHAR(30),
	Description VARCHAR(500),
	Available BOOLEAN DEFAULT TRUE,
	FileType VARCHAR(10) NOT NULL,
	Link VARCHAR(500) NOT NULL,
	Size INT,
	PRIMARY KEY(ProductID),
	FOREIGN KEY(StoreID) REFERENCES Store(StoreID) ON DELETE SET NULL ON UPDATE CASCADE
);
CREATE TABLE ProductImage(
	ProductID INT NOT NULL,
	ProductImageID INT NOT NULL,
	Image VARCHAR(30) NOT NULL,
	PRIMARY KEY(ProductImageID),
	FOREIGN KEY(ProductID) REFERENCES Product(ProductID) ON DELETE SET NULL ON UPDATE CASCADE
);
CREATE TABLE ProductKeyword(
	ProductID INT NOT NULL,
	KeywordID INT NOT NULL,
	Keyword VARCHAR(30) NOT NULL,
	PRIMARY KEY(KeywordID),
	FOREIGN KEY(ProductID) REFERENCES Product(ProductID) ON DELETE SET NULL ON UPDATE CASCADE
);
CREATE TABLE ProductReview(
	ProductID INT NOT NULL,
	UserID INT NOT NULL,
	ReviewID INT NOT NULL,
	Review VARCHAR(500),
	PRIMARY KEY(ReviewID),
	FOREIGN KEY(UserID) REFERENCES User(UserID) ON DELETE SET NULL ON UPDATE CASCADE,
	FOREIGN KEY(ProductID) REFERENCES Product(ProductID) ON DELETE SET NULL ON UPDATE CASCADE
);
CREATE TABLE AnOrder(
	UserID INT NOT NULL,
	ProductID INT NOT NULL,
	OrderID INT NOT NULL,
	Email VARCHAR(30) NOT NULL,
	AmountPaid INT NOT NULL,
	Date DATE NOT NULL,
	PaymentType VARCHAR(15) NOT NULL,
	PRIMARY KEY(OrderID),
	FOREIGN KEY(UserID) REFERENCES User(UserID) ON DELETE SET NULL ON UPDATE CASCADE,
	FOREIGN KEY(ProductID) REFERENCES Product(ProductID) ON DELETE SET NULL ON UPDATE CASCADE
);
CREATE TABLE Message(
	Sender INT NOT NULL,
	Receiver INT NOT NULL,
	MessageID INT NOT NULL,
	Subject VARCHAR(30),
	Text VARCHAR(500),
	Attachment VARCHAR(30),
	WasRead BOOLEAN DEFAULT FALSE,
	Time DATETIME NOT NULL,
	PRIMARY KEY(MessageID),
	FOREIGN KEY(Sender) REFERENCES User(UserID) ON DELETE SET NULL ON UPDATE CASCADE,
	FOREIGN KEY(Receiver) REFERENCES User(UserID) ON DELETE SET NULL ON UPDATE CASCADE
);
